CSS Grid Cheatsheet

Style

display: grid;
grid-template-columns: 100px 100px 100px 100px; (4 column layout with 100px each)

Output

1
2
3
4


Style

display: grid;
grid-template-columns: repeat(4, 200px); (shorthand notation)

Output

1
2
3
4


Style

display: grid;
grid-template-columns: 100px 200px; (2 column layout, one 100px and other 200px. Since there are 4 divs, 2 are pushed to the next row)

Output

1
2
3
4


Style

display: grid;
grid-template-columns: 100px 200px 300px; (3 column layout with varying widths, Since there are 4 divs, last is pushed to the next row)

Output

1
2
3
4


Style

display: grid;
grid-template-columns: 1fr 1fr; (2 column layout, equal proportion, 4 divs)

Output

1
2
3
4


Style

display: grid;
grid-template-columns: 1fr 1fr 1fr 1fr; (4 column layout, equal proporation)

Output

1
2
3
4


Style

display: grid;
grid-template-columns: 1fr 1.5fr 2fr; (3 column layout, varying width)
grid-template-rows: 200px 150px; (1st row height 200px, 2nd row height 150px)

Output

1
2
3
4


Style

display: grid;
grid-template-columns: 1fr 1fr 1fr; (3 column layout, equal proporation)
grid-auto-rows: 100px; (use this when number of rows are unknown, all rows are of height 100px)
grid-gap: 10px;

Output



Style

display: grid;
grid-template-columns: 1fr 1fr 1fr; (3 column layout, equal proporation)
grid-auto-rows: minmax(30px,auto); (use this when row height has to be min 30px)
